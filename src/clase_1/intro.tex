\begin{frame}{Motivación}

    \begin{block}{¿Por qué aprender Docker?}
        \begin{itemize}
            \item Herramienta clave para la vida profesional: reproducir entornos de desarrollo y producción sin dolores de cabeza.
            \pause
            \item No siempre es obligatoria en la currícula (aunque se esta empezando a incluir), pero es altamente esperada en la industria y un poco en investigación.
            \pause
            \item Nos ahorra tiempo: tareas repetitivas como instalar dependencias o configurar entornos pueden automatizarse fácilmente.
    \end{itemize}
\end{block}
    
\end{frame}

\begin{frame}{Beneficios de Docker}
    \begin{block}{Solución}
        \begin{itemize}
            \item En lugar de instalar manualmente todas las dependencias en nuestra máquina, Docker permite levantar un entorno aislado con todo preconfigurado y listo para usar.
            \item Cada componente se ejecuta en su propio contenedor, aislado, y se comunica de manera controlada con los demás servicios.
        \end{itemize}
    \end{block}
\end{frame}

    \begin{frame}
        \frametitle{Imágenes}
        \framesubtitle{Definición}
    
        \begin{itemize}
            \item Imágen de VM (VM Image): \pause Es un archivo que contiene un disco virtual que tiene instalado un sistema operativo, que el hipervisor la arranca como si fuese un servidor físico. \pause
            \item Imagen de container (Container Image): \pause Es es un archivo que incluye todos los archivos, binarios, bibliotecas y configuraciones, para poder ejecutar un contenedor a través de un container engine. \pause
        \end{itemize}
        \begin{block}{Imagen}
            Una imagen puede tener:
            \begin{itemize}
                \item SO
                \item Software
                \item App
            \end{itemize}
        \end{block}
    \end{frame}

\begin{frame}
    \frametitle{¿Cómo podemos compartir nuestras imágenes?}
    \framesubtitle{Registry}

    \begin{block}{Registry de contenedores}
        Un \textbf{registry} es un servicio donde podemos:
        \begin{itemize}
            \item Almacenar imágenes de contenedores.
            \item Compartirlas con otros usuarios o equipos.
            \item Versionarlas y administrarlas de forma centralizada.
        \end{itemize}
        Así como un repositorio de código guarda programas, 
        un registry guarda las imágenes listas para ejecutar.
    \end{block}
\end{frame}

    \frametitle{Motor de contenedores}
    \framesubtitle{Almacenamiento de imágenes}

    \begin{block}{}
        \begin{itemize}
            \item docker run \<image\_name\> \textit{\# Crea y ejecuta un contenedor a partir de una imagen.}
            \item docker start|stop \<container\_name\> \textit{\# Inicia o para un container.}
            \item docker rm \<container\_name\> \textit{\# Remueve el contenedor.}
            \item docker ps \textit{\# Lista los contenedores que están ejecutando.}
            \item docker container stats \textit{\# Muestra el estado del uso de los recursos.}
            \item docker exec \<container\_name\> \<command\> \textit{\# Ejecuta el comando dentro del contenedor.}
            \item docker logs \<container\_name\> \textit{\# Muestra los logs del contenedor.}
        \end{itemize}
    \end{block}

\begin{frame}
    \frametitle{¿Cómo funcionan los volúmenes en Docker?}
    \framesubtitle{Persistencia de datos}

    \begin{block}{Definición}
        Un \textbf{volumen} es un mecanismo de Docker para almacenar datos 
        fuera del ciclo de vida de un contenedor.  
        Sirven para que la información \textbf{no se pierda} al detener o eliminar un contenedor.
    \end{block}

    \begin{itemize}
        \item Los datos se almacenan en el host (por defecto en \texttt{/var/lib/docker/volumes/}).
        \item Se pueden compartir entre varios contenedores.
        \item Separan la \textbf{aplicación} (contenedor) de sus \textbf{datos}.
    \end{itemize}

    \pause
    \begin{block}{Ejemplo}
        \texttt{docker run -v ~/data:/app/data myapp}
        
        \begin{itemize}
            \item \texttt{~/data} → carpeta en el host.
            \item \texttt{/app/data} → carpeta dentro del contenedor.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Variables de entorno en Docker}
    \framesubtitle{Parametrización de contenedores}

    \begin{block}{¿Qué es una variable de entorno?}
        Una \textbf{variable de entorno} es un par \texttt{NOMBRE=VALOR} 
        que se almacena en el sistema operativo y que las aplicaciones 
        pueden leer para modificar su comportamiento sin cambiar el código.
    \end{block}

    \begin{itemize}
        \item Permiten parametrizar la configuración de programas.
        \item Ejemplo típico: base de datos, puerto, modo debug, usuario, contraseña.
        \item Se pueden definir al crear un contenedor y son accesibles desde dentro del mismo.
    \end{itemize}

    \pause
    \begin{block}{Uso en Docker}
        \texttt{docker run -e NOMBRE\_VAR=valor imagen} \\
        Ejemplo: \texttt{docker run -e MYSQL\_ROOT\_PASSWORD=secret mysql}
        
        \begin{itemize}
            \item \texttt{MYSQL\_ROOT\_PASSWORD} es la variable de entorno.
            \item \texttt{secret} es el valor que se le asigna.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{¿Para qué sirven las variables de entorno en Docker?}
    \framesubtitle{Parametrización de contenedores por el usuario}

    \begin{block}{Idea principal}
        Las variables de entorno permiten al usuario \textbf{configurar un contenedor} 
        sin necesidad de modificar la imagen ni el código de la aplicación.
    \end{block}

    \begin{itemize}
        \item Configurar contraseñas, nombres de usuario, puertos o rutas de archivos.
        \item Activar o desactivar modos de operación (ej. modo debug).
        \item Permiten que la misma imagen funcione en distintos entornos 
              (desarrollo, testing, producción) solo cambiando las variables.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{¿Qué es Docker Compose?}
    \framesubtitle{Automatización de contenedores}

    \begin{block}{Definición}
        \textbf{Docker Compose} es una herramienta que permite definir y ejecutar 
        aplicaciones multicontenedor mediante un archivo de configuración (\texttt{docker-compose.yml}).
    \end{block}

    \begin{itemize}
        \item Permite describir contenedores, redes, volúmenes y variables de entorno de forma declarativa.
        \item Sustituye la necesidad de escribir múltiples comandos \texttt{docker run} a mano.
        \item Facilita levantar, detener y escalar aplicaciones completas con un solo comando:
              \texttt{docker-compose up} y \texttt{docker-compose down}.
        \item Mantiene toda la configuración de la aplicación en un solo archivo, fácil de versionar y compartir.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{¿Cómo funcionan los networks en Docker?}
    \framesubtitle{Conexión entre contenedores}

    \begin{block}{Definición}
        Un \textbf{network} en Docker es una red virtual que conecta contenedores 
        entre sí y con el host, de manera aislada y controlada.
    \end{block}

    \begin{itemize}
        \item \textbf{bridge} (por defecto): cada contenedor recibe una IP propia 
              y puede comunicarse con otros usando \textbf{DNS interno de Docker}, 
              es decir, por \textbf{nombre de contenedor} en lugar de IP.
        \item \textbf{host}: el contenedor comparte directamente la red del host 
              (no tiene IP separada).
        \item \textbf{none}: sin acceso a red, solo \texttt{localhost}.
    \end{itemize}

    \pause
    \begin{block}{Ejemplo}
        \texttt{docker network create mi\_red} \\
        \texttt{docker run -d --name web --network mi\_red nginx} \\
        \texttt{docker run -it --rm --network mi\_red alpine ping web}
    \end{block}
\end{frame}
